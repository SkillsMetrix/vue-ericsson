new Vue({
    el: '#app',
    // contains the data wana show to the client
    data:{
        title: 'THIS IS VUE Instance',
        showParagraph: false
    },
    //get calls everytime when invoked
    methods: {
        show: function(){
            this.showParagraph=true
            this.updateTitle('the New VueJS (Updated Data)')
        },
        updateTitle:function(title){
            this.title=title
        }
    },
    // get called only when chenges are detected
    computed:{
        lowercaseTitle:function(){
            return this.title.toLowerCase()
        }
    },
    // uses the function name same as property name,and called when changes detected
    watch:{
        title:function(value){
            alert('Title changed ,new value is ' + value)
        }
    }
})
